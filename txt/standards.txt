position: tuple
    position[0]: int = x-coordinate
        1 = a
        2 = b
        3 = c
        4 = d
        5 = e 
        6 = f 
        7 = g 
        8 = h 
    position[1]: int = y-coordinate

colour: bool
    False = white
    True = black

value: int = value of piece
    1 = pawn
    3 = bishop or knight 
    5 = rook 
    9 = queen 
    10 = king (not legally defined to be 10 but this might be useful)

piece representation:
    lowercase = white
    uppercase = black
    p = pawn
    b = bishop
    n = knight 
    r = rook
    q = queen
    k = king 

disam: int = level of disambiguity required
    0 = none 
    1 = file of departure
    2 = rank of departure
    3 = both file and rank
